// データベース作成メソッド

@Override

    sb_cash.append("CREATE TABLE cash(");
    sb_cash.append("id INTEGER PRIMARY KEY AUTOINCREMENT,");
    sb_cash.append("date TEXT NOT NULL,");

    sb_cash.append("category TEXT NOT NULL,");
    sb_cash.append("memo TEXT");

    sb_emoney.append("memo TEXT");
    sb_emoney.append(");");
